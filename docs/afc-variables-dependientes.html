4.1 AFC variables dependientes


```r
data_tt = data1 %>% select(i_gen1,i_gen2,i_gen5,i_etn1,i_etn2,i_etn3,i_etn5,i_hom1,i_hom2,i_hom3,i_hom4,i_hom5,idschool) %>% as.data.frame()
model_tt <- '
f.genero =~ i_gen1 + i_gen2 + i_gen5 

f.etnico =~ i_etn1 + i_etn2 + i_etn3 + i_etn5

f.homosexual =~ i_hom1 + i_hom2 + i_hom3 + i_hom4 + i_hom5

f.genero ~~ f.etnico 
   f.genero ~~ f.homosexual
   f.etnico ~~ f.homosexual'

fit_tt<- cfa(model_tt, data=data_tt, ordered=c("i_gen1","i_gen2","i_gen5","i_etn1","i_etn2","i_etn3","i_etn5","i_hom1","i_hom2","i_hom3","i_hom4","i_hom5"), cluster = 'idschool')
summary(fit_tt, fit.measures=T, standardized=T)
```
